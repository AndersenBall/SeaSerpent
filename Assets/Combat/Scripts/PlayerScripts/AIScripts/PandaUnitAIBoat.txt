tree("Root")
	FallBack
		tree("Gunner")
		tree("FireCannons")
		tree("PrepFireCannons")
		tree("ReloadCannons")
		tree("RotateCannons")
		tree("AdjustSail")
		tree("RepairShip")
		tree("SteerShip")
		tree("Idle")

tree("Idle")
	while
		ActionCheck("Idle")
	    sequence
			SetWanderPoint
			Idle

tree("Gunner")
	while
		ActionCheck("Gunner")
		selector
			sequence
				IsTrue("BroadsideMode")
				tree("GunnerBroadside")
			sequence
				IsFalse("BroadsideMode")
				tree("GunnerCycle")

tree("GunnerCycle")
	sequence
		FindClosestCannon("Fire")
		GoToClosest("Fire")
		RotateCannon
		FireCannon("False")
		VariableWait("FireCannon")
		tree("ReloadCannon")

tree("ReloadCannon")
	sequence
		FindClosestCannonBall
		GoToClosest("CannonBall")
		GetCannonBall
		Wait(.25)
		GoToClosest("Reload")
		VariableWait("ReloadCannon")
		Reload
		
tree("GunnerBroadside")
    sequence
        tree("Broadside_FireAll")
        tree("ReloadAll")

tree("Broadside_FireAll")
	while
		HasAnyTargetableCannon("Fire")
		sequence
			FindClosestCannon("Fire")
			GoToClosest("Fire")
			RotateCannon
			FireCannon("True")
			VariableWait("FireCannon")
			
tree("PrepFireCannons")
    while
        ActionCheck("PrepCannons")
        sequence
            FindClosestCannon("Fire")
            GoToClosest("Fire")
            
tree("ReloadAll")
	while
		ActionCheck("ReloadCannons")
		sequence
			FindClosestCannonBall
			GoToClosest("CannonBall")
			GetCannonBall
			Wait(.25)
			FindClosestCannon("Reload")
			GoToClosest("Reload")
			VariableWait("ReloadCannon")
			Reload

tree("RotateCannons")
	while
		ActionCheck("RotateCannons")
		sequence
			FindClosestCannon("Rotate")
			GoToClosest("Rotate")
			VariableWait("RotateCannon")
			RotateCannon
			
tree("AdjustSail")
	while
		ActionCheck("AdjustSail")
		sequence
			GoToClosest("sail")
			AdjustSail
tree("RepairShip")
	while
		ActionCheck("Repair")
		sequence
			GoToClosest("hole")
			Repair
tree("SteerShip")
	while
		ActionCheck("steer")
		sequence
			GoToClosest("steeringWheel")
			Steer
