tree("Root")
	FallBack
		tree("Gunner")
		tree("ReloadAllCannons")
		tree("AdjustSail")
		tree("RepairShip")
		tree("SteerShip")
		tree("Idle")

tree("Idle")
	while
		ActionCheck("Idle")
		SetWanderPoint

tree("Gunner")
	while
		ActionCheck("Gunner")
		Sequence
			tree("GunnerBroadside")
			tree("GunnerCycle")

tree("GunnerCycle")
	sequence
		GetBroadsideMode("normal")
		FindClosestCannon("Fire")
		GoToClosest("Fire")
		RotateCannon
		FireCannon("False")
		VariableWait("FireCannon")
		tree("ReloadCannon")
		
tree("GunnerBroadside")
	sequence
		GetBroadsideMode("broadside")
		tree("Broadside_FireAll")
		tree("ReloadAllCannons")

tree("Broadside_FireAll")
	while
		HasAnyTargetableCannon("Fire")
		sequence
			FindClosestCannon("Fire")
			GoToClosest("Fire")
			RotateCannon
			FireCannon("True")
			VariableWait("FireCannon")

tree("ReloadAllCannons")
	while
		HasAnyTargetableCannon("Reload")
		tree("ReloadCannon")
		
tree("ReloadCannon")
	sequence
		FindClosestCannonBall
		GoToClosest("CannonBall")
		GetCannonBall
		Wait(.25)
		FindClosestCannon("Reload")
		GoToClosest("Reload")
		VariableWait("ReloadCannon")
		Reload

tree("PrepFireCannons")
    while
        ActionCheck("PrepCannons")
        sequence
            FindClosestCannon("Fire")
            GoToClosest("Fire")

tree("RotateCannons")
	while
		ActionCheck("RotateCannons")
		sequence
			FindClosestCannon("Rotate")
			GoToClosest("Rotate")
			VariableWait("RotateCannon")
			RotateCannon
			
tree("AdjustSail")
	while
		ActionCheck("AdjustSail")
		sequence
			GoToClosest("sail")
			AdjustSail
tree("RepairShip")
	while
		ActionCheck("Repair")
		sequence
			GoToClosest("hole")
			Repair
tree("SteerShip")
	while
		ActionCheck("steer")
		sequence
			GoToClosest("steeringWheel")
			Steer
