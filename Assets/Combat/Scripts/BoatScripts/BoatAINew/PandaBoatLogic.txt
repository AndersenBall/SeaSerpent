tree("DriveBy")
	sequence
		ActionCheck("DriveBy")
		SetCannonsNeutral
		fallback
			while 
				not GetInAttackPosition(100.0)
				repeat	
					sequence
						ChooseAttackDirection
						CreateAttackVector
						Wait 1.0
			sequence
				UnitsToCannons
				CheckToFire
				Fire
				CheckToRetreat
