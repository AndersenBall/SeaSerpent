tree("Root")
	FallBack
		tree("Dead")
		tree("Player")
		tree("RunAway")
		tree("Attack")

tree("Dead")
	while
		isDead
		Die

tree("Player")
	while
		playerOnBoard
		Idle
		
tree("RunAway")
	sequence
		ActionCheck("RunAway")
		ChooseRunAwayLocation
		Wait 10.0

tree("Attack")
	while
		IsAttacking
		sequence
			ChooseEnemy
			fallback
				tree("DriveBy")
				tree("ApproachTurnShoot")
				tree("Ram")
				tree("FireAtWill")

tree("DriveBy")
	sequence
		ActionCheck("DriveBy")
		SetCannonsNeutral
		DriveBeside
		UnitsToCannons
		CheckToFire
		Fire

tree("ApproachTurnShoot")
	sequence
		ActionCheck("ApproachTurnShoot")
		GetInAttackPosition(100.0)
		TurnToFire
		Wait 10.0

tree("Ram")
	sequence
		ActionCheck("Ram")
		GetInAttackPosition(100.0)
		RamSpeed

tree("FireAtWill")
	sequence
		ActionCheck("FireAtWill")
		TurnToFire
		Wait 10.0
