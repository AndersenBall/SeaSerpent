tree("DriveBy")
	sequence
		ActionCheck("DriveBy")
		fallback
			while 
				not GetInAttackPosition(100.0)
				repeat	
					sequence
						ChooseAttackDirection
						CreateAttackVector
						Wait 1.0	
			sequence
				CheckToFire	
				Fire
				CheckToRetreat
